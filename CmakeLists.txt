cmake_minimum_required(VERSION 3.16...3.30)
set(CMAKE_POLICY_VERSION_MINIMUM 3.5)
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)
set(CMAKE_AUTOUIC ON)

project(SteamUI VERSION 0.1 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(Qt6 REQUIRED COMPONENTS Gui Qml Quick QuickControls2 Concurrent Core)
qt_standard_project_setup()

include(FetchContent)

# 1. Próbujemy znaleźć CURL zainstalowany w MSYS2
find_package(CURL)

# 2. nlohmann/json
set(JSON_BuildTests OFF CACHE INTERNAL "")
FetchContent_Declare(json URL https://github.com/nlohmann/json/releases/download/v3.11.2/json.tar.xz)
FetchContent_MakeAvailable(json)

# 3. cpr - modyfikacja
FetchContent_Declare(cpr 
    GIT_REPOSITORY https://github.com/libcpr/cpr.git 
    GIT_TAG 1.10.x
)
# Jeśli znaleźliśmy systemowy CURL, mówimy CPR żeby go użył
if(CURL_FOUND)
    set(CPR_FORCE_USE_SYSTEM_CURL ON CACHE BOOL "" FORCE)
endif()

FetchContent_MakeAvailable(cpr)

# Aplikacja
qt_add_executable(appSteamUI
    include/steamgrid.h
    source/main.cpp
    source/steamgrid.cpp
    
)

target_include_directories(appSteamUI PRIVATE include)

qt_add_qml_module(appSteamUI
    URI SteamApp
    VERSION 1.0
    QML_FILES 
        ui/Main.qml
        ui/Theme.qml
        ui/CsButton.qml
        ui/Settings.qml
        ui/Progress.qml
)

target_link_libraries(appSteamUI PRIVATE 
    Qt6::Gui 
    Qt6::Qml 
    Qt6::Quick
    Qt6::QuickControls2
    Qt6::Concurrent 
    Qt6::Core # <--- DODAJ TO
    cpr::cpr
    nlohmann_json::nlohmann_json
)
